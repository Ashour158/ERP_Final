# Stage 1: Build the Vite app
FROM node:20-alpine AS build
WORKDIR /app

# Install dependencies (ci if lockfile exists; otherwise install)
COPY package*.json ./
RUN if [ -f package-lock.json ]; then npm ci; else npm install; fi

# Copy source and build
COPY . .
# Allow DigitalOcean to override at build time via Build-time Env
ARG VITE_API_BASE_URL=/api
ENV VITE_API_BASE_URL=$VITE_API_BASE_URL
RUN npm run build

# Stage 2: Serve with Nginx, listening on $PORT (required by DO App Platform)
FROM nginx:1.27-alpine
# Copy built assets
COPY --from=build /app/dist /usr/share/nginx/html
# Template will bind to $PORT at runtime
COPY nginx.conf.template /etc/nginx/templates/default.conf.template
# Official nginx image will envsubst templates from /etc/nginx/templates -> /etc/nginx/conf.d
ENV NGINX_ENVSUBST_OUTPUT_DIR=/etc/nginx/conf.d
# Default for local runs; DO sets PORT automatically
ENV PORT=8080
EXPOSE 8080
